                  [10, 11, 19, ('C', 'C', 'H')], [16, 10, 17, ('H', 'C', 'H')]]
        dihedrals = [[8, 9, 10, 16, ('C', 'O', 'C', 'H')],
                     [8, 9, 10, 17, ('C', 'O', 'C', 'H')],
                     [15, 8, 9, 10, ('H', 'C', 'O', 'C')],
                     [9, 10, 11, 12, ('O', 'C', 'C', 'O')],
                     [9, 10, 11, 18, ('O', 'C', 'C', 'H')],

                     [10, 11, 12, 13, ('C', 'C', 'O', 'C')],

                     [16, 10, 11, 18, ('H', 'C', 'C', 'H')]]
        self.assertEqual(len(self.topology.atoms), 44)
